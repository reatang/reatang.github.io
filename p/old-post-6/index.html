CONTENT_LENGTH:“7”

//请求内容类型

CONTENT_TYPE:“application/x-www-form-urlencoded;charset=UTF-8”

//根目录

DOCUMENT_ROOT:"/home/vagrant/Code/web"

//脚本名称

DOCUMENT_URI:"/fetch/data.php"
