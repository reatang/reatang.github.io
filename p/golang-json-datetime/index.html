package main

import(
    "fmt"
    "encoding/json"
    jsondt "github.com/reatang/go-json-datetime"
)

type Data struct {
    CreateTime jsondt.DateTime `json:"create_time"`
}

func main() {
    j1 := `{"create_time": "2024-07-10 18:52:19"}`

    var d Data
    err := json.Unmarshal(j1, &d)
    if err != nil {
        panic(err)
    }

    fmt.Println(d.CreateTime.Format(time.DateTime))
}
